// ESP32 Pin Definitions (Using safe, standard GPIOs)
const int irSensorPin = 4;   // GPIO 4
const int buttonPin = 5;     // GPIO 5 (Good for buttons)

const int greenPin = 18;     // GPIO 18
const int redPin = 19;       // GPIO 19

const int buzzerPin = 21;    // GPIO 21

int sensorState = 0;
int itemCount = 0;

bool objectDetected = false;

unsigned long lastDetectionTime = 0;
const unsigned long timeout = 10000; // 10 seconds timeout

// Button state tracking
bool lastButtonState = HIGH;

void setup() {
  // Initialize Inputs
  pinMode(irSensorPin, INPUT);
  pinMode(buttonPin, INPUT_PULLUP); // ESP32 has internal pullups on GPIO 5

  // Initialize Outputs
  pinMode(greenPin, OUTPUT);
  pinMode(redPin, OUTPUT);
  pinMode(buzzerPin, OUTPUT);

  // Set all outputs to LOW initially
  digitalWrite(greenPin, LOW);
  digitalWrite(redPin, LOW);
  digitalWrite(buzzerPin, LOW);

  // Initialize Serial Monitor (ESP32 works fine at 9600, though 115200 is typical)
  Serial.begin(9600);

  lastDetectionTime = millis();
  Serial.println("--- ESP32 Conveyor System Initialized ---");
}

void loop() {

  // ================= RESET BUTTON =================
  bool buttonState = digitalRead(buttonPin);

  if (lastButtonState == HIGH && buttonState == LOW) {
    // Button pressed down
    itemCount = 0;

    Serial.println("RESET DONE");
    Serial.print("Count: ");
    Serial.println(itemCount);

    delay(200); // Debounce delay
  }

  lastButtonState = buttonState;

  // ================= IR SENSOR =================
  sensorState = digitalRead(irSensorPin);

  if (sensorState == LOW) {
    // ITEM DETECTED
    digitalWrite(greenPin, HIGH);
    digitalWrite(redPin, LOW);
    digitalWrite(buzzerPin, LOW);

    lastDetectionTime = millis(); // Refresh timeout tracker

    if (!objectDetected) {
      itemCount++;
      Serial.print("Count: ");
      Serial.println(itemCount);
      objectDetected = true;
    }

  } else {
    // NO ITEM DETECTED
    objectDetected = false;

    if (millis() - lastDetectionTime < timeout) {
      // Within 10 seconds window: Keep everything off
      digitalWrite(greenPin, LOW);
      digitalWrite(redPin, LOW);
      digitalWrite(buzzerPin, LOW);
    } else {
      // NO ITEM > 10s: Trigger Alert!
      digitalWrite(greenPin, LOW);
      digitalWrite(redPin, HIGH);
      digitalWrite(buzzerPin, HIGH);
    }
  }

  delay(50); // Small loop stabilization delay
}
